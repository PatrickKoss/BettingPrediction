# dump data from active database
1. python manage.py dumpdata > db.json
2. python manage.py migrate
3. python manage.py shell
# then the shell opens
>> from django.contrib.contenttypes.models import ContentType
>> ContentType.objects.all().delete()
>> exit()
# load data
5. python manage.py loaddata db.json


# for docker-compose
1. docker-compose exec bettingrestapi python manage.py dumpdata > db.json
2. docker-compose exec bettingrestapi python manage.py migrate
3. docker-compose exec bettingrestapi python manage.py shell
# then the shell opens
>> from django.contrib.contenttypes.models import ContentType
>> ContentType.objects.all().delete()
>> exit()
5. docker-compose exec bettingrestapi python manage.py loaddata db.json
